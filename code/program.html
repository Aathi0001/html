Reusable Input Validation Function

This function repeatedly asks the user for input until the input matches a given regular expression. It also supports:

  • Custom prompt message
  • Error message
  • Maximum attempts limit
  • Exit word to terminate input

1. Perl Implementation


sub get_valid_input {
    my ($regex, $prompt, $error, $max_attempts, $exit_word) = @_;

    $exit_word //= "exit";
    my $attempts = 0;

    while (1) {

        print "$prompt: ";
        my $input = <STDIN>;
        chomp $input;

        if (lc($input) eq lc($exit_word)) {
            return undef;
        }

        if ($input =~ $regex) {
            return $input;
        }

        print "$error\n";

        $attempts++;

        if (defined $max_attempts && $attempts >= $max_attempts) {
            return undef;
        }
    }
}

my $name = get_valid_input(qr/^[A-Za-z]{3,20}$/, "Enter name", "Invalid name", 3);

if(defined $name){
    print "Valid Name: $name\n";
}else{
    print "Input failed\n";
}

2. Python Implementation


import re

def get_valid_input(regex, prompt, error, max_attempts=None, exit_word="exit"):
    attempts = 0

    while True:
        user_input = input(prompt + ": ")

        if user_input.lower() == exit_word.lower():
            return None

        if re.match(regex, user_input):
            return user_input

        print(error)

        attempts += 1

        if max_attempts is not None and attempts >= max_attempts:
            return None

3. Java Implementation


import java.util.Scanner;
import java.util.regex.Pattern;

public class InputValidator {

    public static String getValidInput(String regex, String prompt, String error,
                                       Integer maxAttempts, String exitWord) {

        Scanner sc = new Scanner(System.in);
        int attempts = 0;

        if (exitWord == null)
            exitWord = "exit";

        while (true) {

            System.out.print(prompt + ": ");
            String input = sc.nextLine();

            if (input.equalsIgnoreCase(exitWord))
                return null;

            if (Pattern.matches(regex, input))
                return input;

            System.out.println(error);

            attempts++;

            if (maxAttempts != null && attempts >= maxAttempts)
                return null;
        }
    }
}

const readline = require("readline");

function getValidInput(regex, prompt, error, maxAttempts = null, exitWord = "exit") {

    const rl = readline.createInterface({
        input: process.stdin,
        output: process.stdout
    });

    let attempts = 0;

    function ask() {
        rl.question(prompt + ": ", (input) => {

            if (input.toLowerCase() === exitWord.toLowerCase()) {
                rl.close();
                return;
            }

            if (regex.test(input)) {
                console.log("Valid Input:", input);
                rl.close();
                return;
            }

            console.log(error);

            attempts++;

            if (maxAttempts !== null && attempts >= maxAttempts) {
                rl.close();
                return;
            }

            ask();
        });
    }

    ask();
}

Date Validation Utilities

These functions help validate dates for common scenarios such as:

  • Checking if a date is in the future
  • Validating a start and end date range
  • Checking if a date lies within a range

Date format used in all examples: DD-MM-YYYY

1. Check Future Date

Returns true if the given date is in the future.

Perl


use Time::Piece;

sub is_future_date {
    my ($date) = @_;

    my $input_date = Time::Piece->strptime($date, "%d-%m-%Y");
    my $today = localtime;

    return $input_date > $today;
}

3. Check Date Within Range

Checks if a given date lies between a start date and end date.

Perl


use Time::Piece;

sub date_within_range {

    my ($date,$start,$end) = @_;

    my $d = Time::Piece->strptime($date,"%d-%m-%Y");
    my $s = Time::Piece->strptime($start,"%d-%m-%Y");
    my $e = Time::Piece->strptime($end,"%d-%m-%Y");

    return ($d >= $s && $d <= $e);
}

1. Date Overlap

Checks if two date ranges overlap.

Perl


use Time::Piece;

sub date_overlap {

    my ($s1,$e1,$s2,$e2) = @_;

    my $start1 = Time::Piece->strptime($s1,"%d-%m-%Y");
    my $end1   = Time::Piece->strptime($e1,"%d-%m-%Y");

    my $start2 = Time::Piece->strptime($s2,"%d-%m-%Y");
    my $end2   = Time::Piece->strptime($e2,"%d-%m-%Y");

    return ($start1 <= $end2 && $start2 <= $end1);
}

2. Future Time Check

Returns true if the given time is in the future.

Perl


sub is_future_time {

    my ($time) = @_;

    my ($h,$m,$s) = split /:/,$time;

    my ($ch,$cm,$cs) = (localtime)[2,1,0];

    return ($h*3600 + $m*60 + $s) > ($ch*3600 + $cm*60 + $cs);
}

Age Validation

This function validates a user's date of birth and checks whether the age falls within a specified minimum and maximum range.

  • Validates date format (DD-MM-YYYY)
  • Checks invalid calendar ranges
  • Prevents future date of birth
  • Calculates current age
  • Validates minimum age
  • Validates maximum age

1. Perl Implementation


sub validate_age {

    my ($dob,$min_age,$max_age) = @_;

    return "Invalid date of birth"
        unless $dob =~ /^(\d{2})-(\d{2})-(\d{4})$/;

    my ($day,$month,$year) = ($1,$2,$3);

    return "Invalid date of birth"
        if ($day < 1 || $day > 31 || $month < 1 || $month > 12);

    my ($c_day,$c_month,$c_year) = (localtime)[3,4,5];

    $c_year += 1900;
    $c_month += 1;

    if ($year > $c_year ||
       ($year == $c_year && $month > $c_month) ||
       ($year == $c_year && $month == $c_month && $day > $c_day))
    {
        return "Date of birth cannot be in the future";
    }

    my $age = $c_year - $year;

    if ($c_month < $month || ($c_month == $month && $c_day < $day)) {
        $age--;
    }

    return "Age below minimum" if $age < $min_age;
    return "Age above maximum" if $age > $max_age;

    return "Age valid";
}

import re
from datetime import datetime

def validate_age(dob,min_age,max_age):

    if not re.match(r"^\d{2}-\d{2}-\d{4}$",dob):
        return "Invalid date of birth"

    day,month,year = map(int,dob.split("-"))

    if day < 1 or day > 31 or month < 1 or month > 12:
        return "Invalid date of birth"

    today = datetime.today()

    if datetime(year,month,day) > today:
        return "Date of birth cannot be in the future"

    age = today.year - year

    if (today.month,today.day) < (month,day):
        age -= 1

    if age < min_age:
        return "Age below minimum"

    if age > max_age:
        return "Age above maximum"

    return "Age valid"

import java.time.LocalDate;
import java.time.Period;
import java.time.format.DateTimeFormatter;

public static String validateAge(String dob,int minAge,int maxAge){

    DateTimeFormatter fmt =
        DateTimeFormatter.ofPattern("dd-MM-yyyy");

    LocalDate birthDate = LocalDate.parse(dob,fmt);
    LocalDate today = LocalDate.now();

    if(birthDate.isAfter(today))
        return "Date of birth cannot be in the future";

    int age = Period.between(birthDate,today).getYears();

    if(age < minAge)
        return "Age below minimum";

    if(age > maxAge)
        return "Age above maximum";

    return "Age valid";
}

1. Amount Validation

Validates a monetary amount with checks for:

  • Correct numeric format
  • No negative values
  • No zero amount
  • Decimal limit
  • Minimum amount
  • Maximum amount

Perl


sub validate_amount {

    my ($amount,$min,$max,$decimal_limit) = @_;

    return "Invalid amount format"
        unless $amount =~ /^\d+(\.\d+)?$/;

    return "Negative amount not allowed"
        if $amount < 0;

    return "Zero amount not allowed"
        if $amount == 0;

    if (defined $decimal_limit && $amount =~ /\.(\d+)/) {
        return "Invalid decimal places"
            if length($1) > $decimal_limit;
    }

    if (defined $min) {
        return "Amount below minimum"
            if $amount < $min;
    }

    if (defined $max) {
        return "Amount exceeds maximum"
            if $amount > $max;
    }

    return "Amount valid";
}

2. Luhn Validation (Credit/Debit Card)

The Luhn algorithm validates card numbers used in credit cards, debit cards, and identification numbers.

Perl


sub validate_luhn {

    my ($number) = @_;

    my @digits = reverse split //,$number;
    my $sum = 0;

    for my $i (0..$#digits){

        my $d = $digits[$i];

        if($i % 2 == 1){

            $d *= 2;

            $d -= 9 if $d > 9;
        }

        $sum += $d;
    }

    return ($sum % 10 == 0)
        ? "Valid card number"
        : "Invalid card number";
}

Python


def validate_luhn(number):

        if i%2==1:
            d*=2
            if d>9:
                d-=9

        total+=d

1. File Validation

Validates a file by checking:

  • File existence
  • Read permission
  • Write permission
  • Empty file
  • Maximum size
  • Allowed file types

Perl


sub validate_file {

    my ($file,$max_size,$allowed_types)=@_;

    return "File not found"
        unless -e $file;

    return "File unreadable"
        unless -r $file;

    return "File not writable"
        unless -w $file;

    return "File empty"
        if -z $file;

    if(defined $max_size){
        my $size=-s $file;

        return "File size too large"
            if $size > $max_size;
    }

    if(defined $allowed_types){

        my ($ext)=$file =~ /\.([^.]+)$/;

        return "Invalid file type"
            unless grep {$_ eq $ext} @$allowed_types;
    }

    return "File valid";
}

Python


import os

def validate_file(file,max_size=None,allowed_types=None):

    if not os.path.exists(file):
        return "File not found"

    if not os.access(file,os.R_OK):
        return "File unreadable"

    if not os.access(file,os.W_OK):
        return "File not writable"

    if os.path.getsize(file)==0:
        return "File empty"

    if max_size and os.path.getsize(file)>max_size:
        return "File size too large"

    if allowed_types:
        ext=file.split(".")[-1]

        if ext not in allowed_types:
            return "Invalid file type"

    return "File valid"

Java


import java.io.File;

public static String validateFile(
String path,
Long maxSize,
String[] allowedTypes){

    File file=new File(path);

    if(!file.exists())
        return "File not found";

    if(!file.canRead())
        return "File unreadable";

    if(!file.canWrite())
        return "File not writable";

    if(file.length()==0)
        return "File empty";

    if(maxSize!=null && file.length()>maxSize)
        return "File size too large";

    if(allowedTypes!=null){

        String ext=path.substring(
            path.lastIndexOf(".")+1
        );

        boolean ok=false;

        for(String t:allowedTypes)
            if(t.equals(ext))
                ok=true;

        if(!ok)
            return "Invalid file type";
    }

    return "File valid";
}

PHP


function validate_file($file,$max_size=null,$allowed_types=null){

    if(!file_exists($file))
        return "File not found";

    if(!is_readable($file))
        return "File unreadable";

    if(!is_writable($file))
        return "File not writable";

    if(filesize($file)==0)
        return "File empty";

    if($max_size && filesize($file)>$max_size)
        return "File size too large";

    if($allowed_types){

            return "Invalid file type";
    }

    return "File valid";
}
